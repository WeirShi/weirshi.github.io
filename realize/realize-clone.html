    /* 实现深拷贝函数, 要满足以下要求
 * 1. 支持数组，正则，函数以及JSON支持的数据类型
 * 2. 解决循环引用的问题
 */
function isType(obj) {
  return Object.prototype.toString.call(obj).slice(8, -1)
}
function deepClone(obj) {
  switch (isType(obj)) {
    case 'Null':
    case 'Undefined':
    case 'Symbol':
    case 'String':
    case 'Number':
    case 'Boolean':
      return obj
  }
  const map = new WeakMap()
  const temp = map.get(obj)
  if (temp) {
    return temp
  }

  // 其他包装类型
  switch (isType(obj)) {
    case 'RegExp':
    case 'Date':
    case 'Error':
      return new obj.constructor(obj.valueOf())
  }

  const resObj = Array.isArray(obj)  ? [] : {}
  map.set(obj, resObj)
  // for in 会遍历到原型链上的属性
  for (let k in obj) {
    if (obj.hasOwnProperty(k)) {
      resObj[k] = deepClone(obj[k], map)
    }
  }
  return resObj
}
